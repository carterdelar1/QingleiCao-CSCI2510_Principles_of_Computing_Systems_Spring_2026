9. void printReverse(char* string) {
        int i = 0;
        int string_length;

        while (string[i] != NULL) {
            i++;
        }
        string_length = i;

        for (string_length > 0; string_length--;) {
            printf("%c\n", *(string+string_length));
        }
    };
